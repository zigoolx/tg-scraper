export ALL_PROXY='socks5h://user:pass@ip:port'
export all_proxy="$ALL_PROXY"
export http_proxy="$ALL_PROXY"
export https_proxy="$ALL_PROXY"
export HTTP_PROXY="$ALL_PROXY"
export HTTPS_PROXY="$ALL_PROXY"
export NO_PROXY="localhost,127.0.0.1"

bash <(curl -Ls --proxy "$ALL_PROXY" https://raw.githubusercontent.com/mhsanaei/3x-ui/master/install.sh)

طبق اطلاعاتی که منتشر شده اینترنت قراره توی 4 مرحله وصل بشه
خیلی خلاصه میگم سرتون درد نگیره :
مرحله اول : کاهش محدودیت خدمات داخلی ، مثلا توی پیام رسان هایی مثل ایتا نمیشد ویدیو ارسال کنی یا بیشتر از 5 مگ نمیشد فایل آپلود کنی حالا این محدودیت برداشته میشه.

مرحله دوم : وایت لیست کردن سایت های حیاتی که توی این یکی دو روز اخیر در دسترس قرار گرفتن برای جلوگیری از اختلال در کسب و کار های قانونی.
تو این مرحله اومدن مسیریابی ترافیک vpn رو شناسایی کردن تا بتونند راهای نفوذ رو ببندن

مرحله سوم : گشایش پلتفرم های سرویس هوش مصنوعی با هدف فراهم سازی دسترسی کنترل شده ، یعنی مثلا سایت chatgpt باز میشه ولی از دامنه یا ایپیش نمیتونید برا ساخت کانفیگ استفاده کنید چون اومدن جلوشو بستن.

مرحله چهارم : گشایش تدریجی پروتکل های شبکه که تو این مرحله تقریبا همه چی قابل دسترسیه ولی از اونجایی که روی پروتکل های icmp , ipv6 , udp میشه تانل زد اومدن راهکار دادن گفتن کلا اینا باید به خارج بسته باشن یا محدودیت شدید گذاشته بشه براشون

یه سری سیاست گذاری الزامی هم انجام دادن :
تمامی سایت های آپلودر باید ثبت نام و احراز هویت رو برای کاربر اجباری کنند
هرکسی سایت و خدمات توی ایران داره باید اونو ببره پشت شبکه توزیع محتوا ملی
